def K_Folds(data):
    indexNo = len(data) / 3
    return [data[round(indexNo * i):round(indexNo * (i + 1))] for i in range(3)]

def data_gen(x):
    y=[]
    for i in range(len(x)):
        y.append(3*x[i]*x[i]*x[i]+2*x[i]*x[i]-3*x[i]+1+rand[i])
    return y
